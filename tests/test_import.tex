\documentclass{unittest}
\usepackage{unicode-symbols}


\makeatletter
% \strip@prefix@safe{⟨tokens⟩} returns ⟨tokens⟩ with a leading ">" removed if one is present;
% if no ">" is present at the start, the argument is returned unchanged.
\newcommand{\strip@prefix@safe}[1]{%
\edef\aga{#1>}
\edef\temp{\expandafter\strip@prefix\aga}%
% check if \temp is empty
\ifx\temp\@empty%
<#1>%
\else%
<\temp%
\fi%
% \temp%
}


\NewDocumentCommand{\xmeaning}{m}{% prints the meaning of a unicode symbol
\edef\foo{\meaning#1}
\strip@prefix@safe{\foo}%
% \edef\stripped{\strip@prefix{\foo >}}
% \ifx\stripped\relax%
% stripped%
% \else%
% not
% % \expandafter\strip@prefix\unicode@meaning@tempA%
% \fi%
% \stripped%
}%
\makeatother
